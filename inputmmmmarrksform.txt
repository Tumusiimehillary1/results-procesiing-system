<script>
    let studentCount = 0;

    function addStudent() {
        studentCount++;
        const container = document.getElementById('studentsContainer');

        const studentDiv = document.createElement('div');
        studentDiv.innerHTML = `
            <h3>Student ${studentCount}</h3>
            <label for="student_${studentCount}_name">Student Name:</label>
            <input type="text" id="student_${studentCount}_name" name="student_${studentCount}_name" required>
            <br>
            <label for="student_${studentCount}_math">Math Marks:</label>
            <input type="number" id="student_${studentCount}_math" name="student_${studentCount}_math" required>
            <br>
            <label for="student_${studentCount}_english">English Marks:</label>
            <input type="number" id="student_${studentCount}_english" name="student_${studentCount}_english" required>
            <br>
            <label for="student_${studentCount}_sst">SST Marks:</label>
            <input type="number" id="student_${studentCount}_sst" name="student_${studentCount}_sst" required>
            <br>
            <label for="student_${studentCount}_science">Science Marks:</label>
            <input type="number" id="student_${studentCount}_science" name="student_${studentCount}_science" required>
            <br>
        `;

        container.appendChild(studentDiv);
    }
</script>
